w2 = float(input())
w3 = float(input())
w4 = float(input())

# get soma
soma = float(input())

# Calculate the sum of neuron input and its weight
sum = (a1 * w1) + (a2 * w2) + (a3 * w3) + (a4 * w4)

# Output condition
if sum > soma :
    print(1)

else:
    print(0)
